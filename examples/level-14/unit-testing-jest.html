Level 14: Unit Testing with Jest

Jest Testing Framework Overview

JavaScript testing framework with built-in assertions, mocking, and test runner

Basic Test Structure

// math.js
export function add(a, b) {
  return a + b;
}

export function multiply(a, b) {
  return a * b;
}

// math.test.js
import { add, multiply } from './math';

describe('Math functions', () => {
  test('adds two numbers correctly', () => {
    expect(add(2, 3)).toBe(5);
    expect(add(-1, 1)).toBe(0);
    expect(add(0, 0)).toBe(0);
  });

  test('multiplies two numbers correctly', () => {
    expect(multiply(2, 3)).toBe(6);
    expect(multiply(-2, 3)).toBe(-6);
    expect(multiply(0, 5)).toBe(0);
  });
});

Test Setup and Teardown

describe('User Service', () => {
  let userService;
  let mockApi;

  beforeEach(() => {
    mockApi = {
      get: jest.fn(),
      post: jest.fn(),
      put: jest.fn(),
      delete: jest.fn()
    };
    userService = new UserService(mockApi);
  });

  afterEach(() => {
    jest.clearAllMocks();
  });

  test('fetches user by id', async () => {
    const mockUser = { id: 1, name: 'John Doe' };
    mockApi.get.mockResolvedValue(mockUser);

    const user = await userService.getUser(1);

    expect(mockApi.get).toHaveBeenCalledWith('/users/1');
    expect(user).toEqual(mockUser);
  });
});

Jest Matchers and Assertions

Comprehensive set of matchers for different types of assertions

Equality Matchers

test('equality matchers', () => {
  // Primitive values
  expect(2 + 2).toBe(4);
  expect('hello').toBe('hello');
  expect(true).toBe(true);

  // Object equality
  const user = { name: 'John', age: 30 };
  expect(user).toEqual({ name: 'John', age: 30 });
  expect(user).not.toBe({ name: 'John', age: 30 });

  // Array equality
  expect([1, 2, 3]).toEqual([1, 2, 3]);
  expect([1, 2, 3]).toContain(2);
});

Truthiness Matchers

test('truthiness matchers', () => {
  // Falsy values
  expect(null).toBeNull();
  expect(undefined).toBeUndefined();
  expect(false).toBeFalsy();
  expect(0).toBeFalsy();
  expect('').toBeFalsy();

  // Truthy values
  expect(1).toBeTruthy();
  expect('hello').toBeTruthy();
  expect({}).toBeTruthy();
  expect([]).toBeTruthy();

  // Defined/Undefined
  expect('value').toBeDefined();
  expect(undefined).toBeUndefined();
});

Number Matchers

test('number matchers', () => {
  const value = 2 + 2;

  expect(value).toBeGreaterThan(3);
  expect(value).toBeGreaterThanOrEqual(4);
  expect(value).toBeLessThan(5);
  expect(value).toBeLessThanOrEqual(4);

  // Floating point equality
  expect(0.1 + 0.2).toBeCloseTo(0.3);
  expect(0.1 + 0.2).not.toBe(0.3);
});

String Matchers

test('string matchers', () => {
  const str = 'Hello World';

  expect(str).toMatch(/Hello/);
  expect(str).toMatch('World');
  expect(str).toContain('Hello');
  expect(str).toHaveLength(11);

  // Case insensitive
  expect(str).toMatch(/hello/i);
});

Mocking with Jest

Mock functions, modules, and external dependencies

Function Mocking

// Mock functions
const mockCallback = jest.fn();

test('mock function calls', () => {
  mockCallback('arg1', 'arg2');

  expect(mockCallback).toHaveBeenCalled();
  expect(mockCallback).toHaveBeenCalledWith('arg1', 'arg2');
  expect(mockCallback).toHaveBeenCalledTimes(1);

  // Reset mock
  mockCallback.mockClear();
  expect(mockCallback).not.toHaveBeenCalled();
});

// Mock return values
const mockFn = jest.fn();
mockFn.mockReturnValue('mocked value');
expect(mockFn()).toBe('mocked value');

// Mock implementation
mockFn.mockImplementation((x) => x * 2);
expect(mockFn(5)).toBe(10);

Module Mocking

// Mock entire modules
jest.mock('./api', () => ({
  fetchUser: jest.fn(),
  updateUser: jest.fn(),
}));

// Mock specific functions
jest.mock('./utils', () => ({
  ...jest.requireActual('./utils'),
  formatDate: jest.fn(() => '2023-01-01'),
}));

// Mock with implementation
jest.mock('./logger', () => ({
  log: jest.fn(),
  error: jest.fn(),
}));

import { log, error } from './logger';

test('module mocking', () => {
  log('test message');
  expect(log).toHaveBeenCalledWith('test message');
});

Async Mocking

// Mock async functions
const mockFetch = jest.fn();

mockFetch.mockResolvedValue({
  json: () => Promise.resolve({ data: 'test' })
});

test('async mocking', async () => {
  const response = await mockFetch('/api/data');
  const data = await response.json();

  expect(mockFetch).toHaveBeenCalledWith('/api/data');
  expect(data).toEqual({ data: 'test' });
});

// Mock rejected promises
mockFetch.mockRejectedValue(new Error('Network error'));

test('async error handling', async () => {
  await expect(mockFetch('/api/data')).rejects.toThrow('Network error');
});

Timer Mocking

// Mock timers
jest.useFakeTimers();

test('timer mocking', () => {
  const callback = jest.fn();
  
  setTimeout(callback, 1000);
  
  // Fast-forward time
  jest.advanceTimersByTime(1000);
  
  expect(callback).toHaveBeenCalled();
});

// Mock setInterval
test('interval mocking', () => {
  const callback = jest.fn();
  
  setInterval(callback, 1000);
  
  jest.advanceTimersByTime(3000);
  
  expect(callback).toHaveBeenCalledTimes(3);
});

// Cleanup
afterEach(() => {
  jest.useRealTimers();
});
