React Testing Library Overview

Simple and complete testing utilities for React components

Basic Component Testing

import { render, screen } from '@testing-library/react';
import { Button } from './Button';

test('renders button with text', () => {
  render(<Button>Click me</Button>);
  
  const button = screen.getByRole('button', { name: /click me/i });
  expect(button).toBeInTheDocument();
  expect(button).toHaveTextContent('Click me');
});

test('calls onClick when clicked', () => {
  const handleClick = jest.fn();
  render(<Button onClick={handleClick}>Click me</Button>);
  
  const button = screen.getByRole('button');
  fireEvent.click(button);
  
  expect(handleClick).toHaveBeenCalledTimes(1);
});

Query Methods

// Different query methods
test('query methods', () => {
  render(<MyComponent />);
  
  // getBy - throws if not found
  const heading = screen.getByRole('heading');
  const button = screen.getByText('Submit');
  
  // queryBy - returns null if not found
  const missing = screen.queryByText('Not found');
  expect(missing).toBeNull();
  
  // findBy - for async elements
  const asyncElement = await screen.findByText('Loaded');
  
  // getAllBy - returns array
  const buttons = screen.getAllByRole('button');
  expect(buttons).toHaveLength(3);
});

User Events and Interactions

Testing user interactions with fireEvent and user-event

Form Testing

import { render, screen, fireEvent } from '@testing-library/react';
import userEvent from '@testing-library/user-event';

test('form submission', async () => {
  const user = userEvent.setup();
  const handleSubmit = jest.fn();
  
  render(<ContactForm onSubmit={handleSubmit} />);
  
  // Fill form fields
  await user.type(screen.getByLabelText(/name/i), 'John Doe');
  await user.type(screen.getByLabelText(/email/i), 'john@example.com');
  
  // Submit form
  await user.click(screen.getByRole('button', { name: /submit/i }));
  
  expect(handleSubmit).toHaveBeenCalledWith({
    name: 'John Doe',
    email: 'john@example.com'
  });
});

test('input validation', async () => {
  const user = userEvent.setup();
  render(<ContactForm />);
  
  const emailInput = screen.getByLabelText(/email/i);
  
  // Type invalid email
  await user.type(emailInput, 'invalid-email');
  await user.tab(); // Trigger blur
  
  expect(screen.getByText(/invalid email/i)).toBeInTheDocument();
});

User Interactions

test('user interactions', async () => {
  const user = userEvent.setup();
  render(<InteractiveComponent />);
  
  // Click events
  await user.click(screen.getByRole('button'));
  
  // Double click
  await user.dblClick(screen.getByText('Double click me'));
  
  // Hover
  await user.hover(screen.getByText('Hover me'));
  
  // Keyboard events
  await user.keyboard('{Tab}');
  await user.keyboard('{Enter}');
  await user.keyboard('{Escape}');
  
  // Select options
  await user.selectOptions(screen.getByRole('combobox'), 'option1');
  
  // Check checkboxes
  await user.click(screen.getByRole('checkbox'));
});

Custom Render and Providers

Testing components with context providers and custom render functions

Context Provider Testing

import { render, screen } from '@testing-library/react';
import { ThemeProvider } from './ThemeProvider';
import { UserProvider } from './UserProvider';

// Custom render with providers
const renderWithProviders = (ui, options = {}) => {
  const { theme = 'light', user = null } = options;
  
  return render(
    <ThemeProvider value={theme}>
      <UserProvider value={user}>
        {ui}
      </UserProvider>
    </ThemeProvider>
  );
};

test('renders with theme provider', () => {
  renderWithProviders(<ThemedComponent />, { theme: 'dark' });
  
  expect(screen.getByTestId('themed-component')).toHaveClass('dark');
});

test('renders with user context', () => {
  const mockUser = { name: 'John', email: 'john@example.com' };
  renderWithProviders(<UserProfile />, { user: mockUser });
  
  expect(screen.getByText('John')).toBeInTheDocument();
});

Custom Hooks Testing

import { renderHook, act } from '@testing-library/react';
import { useCounter } from './useCounter';

test('useCounter hook', () => {
  const { result } = renderHook(() => useCounter(0));
  
  expect(result.current.count).toBe(0);
  
  act(() => {
    result.current.increment();
  });
  
  expect(result.current.count).toBe(1);
  
  act(() => {
    result.current.decrement();
  });
  
  expect(result.current.count).toBe(0);
});

test('useCounter with initial value', () => {
  const { result } = renderHook(() => useCounter(5));
  
  expect(result.current.count).toBe(5);
});
