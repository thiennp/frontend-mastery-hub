Route Guards

import { Navigate, useLocation } from 'react-router-dom';

// Protected Route Component
function ProtectedRoute({ children, isAuthenticated, redirectTo = '/login' }) {
  const location = useLocation();
  
  if (!isAuthenticated) {
    return <Navigate to={redirectTo} state={{ from: location }} replace />;
  }
  
  return children;
}

// Usage in App
function App() {
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  
  return (
    <Routes>
      <Route path="/login" element={<Login />} />
      <Route 
        path="/dashboard" 
        element={
          <ProtectedRoute isAuthenticated={isAuthenticated}>
            <DashboardLayout />
          </ProtectedRoute>
        }
      >
        <Route index element={<DashboardHome />} />
        <Route path="users" element={<Users />} />
      </Route>
    </Routes>
  );
}

// Role-based protection
function RoleProtectedRoute({ children, requiredRole, userRole }) {
  if (userRole !== requiredRole) {
    return <Navigate to="/unauthorized" replace />;
  }
  
  return children;
}

        roles: ['admin', 'manager']
      },
      {
        path: 'settings',
        element: <Settings />,
        title: 'Settings',
        roles: ['admin']
      }
    ]
  }
];

// Route configuration renderer
function renderRoutes(routes) {
  return routes.map(route => (
    <Route 
      key={route.path} 
      path={route.path} 
      element={route.protected ? 
        <ProtectedRoute>{route.element}</ProtectedRoute> : 
        route.element
      }
    >
      {route.children && renderRoutes(route.children)}
    </Route>
  ));
}

Breadcrumb Navigation

Dynamic breadcrumb generation and navigation

Breadcrumb Component

import { useLocation, Link } from 'react-router-dom';

function Breadcrumb() {
  const location = useLocation();
  
  const generateBreadcrumbs = () => {
    const pathnames = location.pathname.split('/').filter(x => x);
    const breadcrumbs = [];
    
    let currentPath = '';
    
    pathnames.forEach((name, index) => {
      currentPath += `/${name}`;
      
      breadcrumbs.push({
        name: name.charAt(0).toUpperCase() + name.slice(1),
        path: currentPath,
        isLast: index === pathnames.length - 1
      });
    });
    
    return breadcrumbs;
  };
  
  const breadcrumbs = generateBreadcrumbs();
  
  return (
    <nav className="breadcrumb">
      <Link to="/">Home</Link>
      {breadcrumbs.map((breadcrumb, index) => (
        <span key={breadcrumb.path}>
          {breadcrumb.isLast ? (
            <span className="breadcrumb-current">
              {breadcrumb.name}
            </span>
          ) : (
            <Link to={breadcrumb.path}>
              {breadcrumb.name}
            </Link>
          )}
        </span>
      ))}
    </nav>
  );
}

Custom Breadcrumb Data

// Breadcrumb configuration
const breadcrumbConfig = {
  '/dashboard': 'Dashboard',
  '/dashboard/users': 'Users',
  '/dashboard/settings': 'Settings',
  '/products': 'Products',
  '/products/:id': (params) => `Product ${params.id}`,
  '/products/:id/edit': (params) => `Edit Product ${params.id}`
};

function SmartBreadcrumb() {
  const location = useLocation();
  const params = useParams();
  
  const getBreadcrumbName = (path) => {
    // Check for exact match first
    if (breadcrumbConfig[path]) {
      return typeof breadcrumbConfig[path] === 'function' 
        ? breadcrumbConfig[path](params)
        : breadcrumbConfig[path];
    }
    
    // Check for parameterized routes
    for (const [route, name] of Object.entries(breadcrumbConfig)) {
      if (route.includes(':')) {
        const routePattern = route.replace(/:[^/]+/g, '[^/]+');
        const regex = new RegExp(`^${routePattern}$`);
        if (regex.test(path)) {
          return typeof name === 'function' ? name(params) : name;
        }
      }
    }
    
    // Fallback to path name
    return path.split('/').pop() || 'Home';
  };
  
  const breadcrumbs = location.pathname
    .split('/')
    .filter(x => x)
    .map((_, index, array) => {
      const path = '/' + array.slice(0, index + 1).join('/');
      return {
        name: getBreadcrumbName(path),
        path: path
      };
    });
  
  return (
    <nav className="breadcrumb">
      <Link to="/">Home</Link>
      {breadcrumbs.map((breadcrumb, index) => (
        <span key={breadcrumb.path}>
          {index === breadcrumbs.length - 1 ? (
            <span className="breadcrumb-current">
              {breadcrumb.name}
            </span>
          ) : (
            <Link to={breadcrumb.path}>
              {breadcrumb.name}
            </Link>
          )}
        </span>
      ))}
    </nav>
  );
}
