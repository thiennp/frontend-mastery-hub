Router Setup & Configuration

Understanding different router types and basic setup

BrowserRouter

import { BrowserRouter, Routes, Route } from 'react-router-dom';

function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />
        <Route path="/contact" element={<Contact />} />
      </Routes>
    </BrowserRouter>
  );
}

// Uses HTML5 history API
// Clean URLs: /about
// Requires server configuration for SPA

HashRouter

import { HashRouter, Routes, Route } from 'react-router-dom';

function App() {
  return (
    <HashRouter>
      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />
        <Route path="/contact" element={<Contact />} />
      </Routes>
    </HashRouter>
  );
}

// Uses hash-based routing
// URLs: /#/about
// Works without server configuration
// Good for static hosting

MemoryRouter

import { MemoryRouter, Routes, Route } from 'react-router-dom';

function App() {
  return (
    <MemoryRouter initialEntries={['/about']}>
      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />
        <Route path="/contact" element={<Contact />} />
      </Routes>
    </MemoryRouter>
  );
}

// In-memory routing
// No URL changes
// Perfect for testing
// Good for mobile apps

NavLink Component

import { NavLink } from 'react-router-dom';

function Navigation() {
  return (
    <nav>
      <NavLink 
        to="/" 
        className={({ isActive }) => 
          isActive ? 'nav-link active' : 'nav-link'
        }
      >
        Home
      </NavLink>
      
      <NavLink 
        to="/about"
        style={({ isActive }) => ({
          color: isActive ? 'red' : 'blue'
        })}
      >
        About
      </NavLink>

      <NavLink 
        to="/products" 
        end
        className="nav-link"
      >
        Products
      </NavLink>
    </nav>
  );
}

Programmatic Navigation

import { useNavigate, useLocation } from 'react-router-dom';

function ProductCard({ product }) {
  const navigate = useNavigate();
  const location = useLocation();
  
  const handleViewProduct = () => {
    // Navigate to product detail
    navigate(`/products/${product.id}`);
  };
  
  const handleEditProduct = () => {
    // Navigate with state
    navigate(`/products/${product.id}/edit`, {
      state: { product, from: location.pathname }
    });
  };
  
  const handleGoBack = () => {
    // Go back in history
    navigate(-1);
  };
  
  return (
    <div>
      <h3>{product.name}</h3>
      <button onClick={handleViewProduct}>View</button>
      <button onClick={handleEditProduct}>Edit</button>
      <button onClick={handleGoBack}>Back</button>
    </div>
  );
}

Route Parameters & Query Strings

Dynamic routing with parameters and search queries

Route Parameters

import { Routes, Route, useParams } from 'react-router-dom';

// Route configuration
function App() {
  return (
    <Routes>
      <Route path="/users/:userId" element={<UserProfile />} />
      <Route path="/products/:category/:id" element={<Product />} />
      <Route path="/posts/:year/:month/:slug" element={<BlogPost />} />
    </Routes>
  );
}

// Accessing parameters
function UserProfile() {
  const { userId } = useParams();
  
  return (
    <div>
      <h1>User Profile: {userId}</h1>
    </div>
  );
}

function Product() {
  const { category, id } = useParams();
  
  return (
    <div>
      <h1>Product: {id} in {category}</h1>
    </div>
  );
}

Query Parameters

import { useSearchParams, useLocation } from 'react-router-dom';

function ProductList() {
  const [searchParams, setSearchParams] = useSearchParams();
  const location = useLocation();
  
  // Get query parameters
  const category = searchParams.get('category');
  const sort = searchParams.get('sort');
  const page = searchParams.get('page');
  
  // Update query parameters
  const handleCategoryChange = (newCategory) => {
    setSearchParams(prev => {
      prev.set('category', newCategory);
      prev.set('page', '1'); // Reset page
      return prev;
    });
  };
  
  const handleSortChange = (newSort) => {
    setSearchParams(prev => {
      prev.set('sort', newSort);
      return prev;
    });
  };
  
  // Get all search params as object
  const allParams = Object.fromEntries(searchParams);
  
  return (
    <div>
      <h1>Products</h1>
      <p>Category: {category}</p>
      <p>Sort: {sort}</p>
      <p>Page: {page}</p>
      <p>All params: {JSON.stringify(allParams)}</p>
      
      <select onChange={(e) => handleCategoryChange(e.target.value)}>
        <option value="electronics">Electronics</option>
        <option value="clothing">Clothing</option>
      </select>
    </div>
  );
}

Optional Parameters

import { Routes, Route, useParams } from 'react-router-dom';

function App() {
  return (
    <Routes>
      {/* Optional parameter with ? */}
      <Route path="/users/:userId?" element={<UserList />} />
      
      {/* Multiple optional parameters */}
      <Route path="/search/:query?/:category?" element={<Search />} />
      
      {/* Wildcard routes */}
      <Route path="/files/*" element={<FileBrowser />} />
    </Routes>
  );
}

function UserList() {
  const { userId } = useParams();
  
  if (userId) {
    return <UserProfile userId={userId} />;
  }
  
  return <UserListComponent />;
}

function Search() {
  const { query, category } = useParams();
  
  return (
    <div>
      <h1>Search Results</h1>
      {query && <p>Query: {query}</p>}
      {category && <p>Category: {category}</p>}
    </div>
  );
}
