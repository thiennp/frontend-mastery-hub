Level 15: React Optimization Techniques

React.memo for Preventing Re-renders

Optimize components by preventing unnecessary re-renders

Basic React.memo

import React, { memo } from 'react';

// Without memo - re-renders on every parent update
function ExpensiveComponent({ data, onUpdate }) {
  console.log('ExpensiveComponent rendered');
  
  return (
    <div>
      <h3>Data: {data}</h3>
      <button onClick={onUpdate}>Update</button>
    </div>
  );
}

// With memo - only re-renders when props change
const OptimizedComponent = memo(function ExpensiveComponent({ data, onUpdate }) {
  console.log('OptimizedComponent rendered');
  
  return (
    <div>
      <h3>Data: {data}</h3>
      <button onClick={onUpdate}>Update</button>
    </div>
  );
});

// Custom comparison function
const CustomMemoComponent = memo(
  function ExpensiveComponent({ data, onUpdate }) {
    return (
      <div>
        <h3>Data: {data}</h3>
        <button onClick={onUpdate}>Update</button>
      </div>
    );
  },
  (prevProps, nextProps) => {
    // Only re-render if data changes, ignore onUpdate
    return prevProps.data === nextProps.data;
  }
);

useMemo Hook

import React, { useState, useMemo } from 'react';

function ExpensiveCalculation({ items, multiplier }) {
  // Expensive calculation that only runs when dependencies change
  const expensiveValue = useMemo(() => {
    console.log('Expensive calculation running...');
    return items.reduce((sum, item) => sum + item.value, 0) * multiplier;
  }, [items, multiplier]);

  return (
    <div>
      <h3>Expensive Value: {expensiveValue}</h3>
      <p>Items: {items.length}</p>
    </div>
  );
}

function MemoizedList({ items, filter }) {
  // Memoize filtered items to prevent re-filtering
  const filteredItems = useMemo(() => {
    console.log('Filtering items...');
    return items.filter(item => 
      item.name.toLowerCase().includes(filter.toLowerCase())
    );
  }, [items, filter]);

  return (
    <ul>
      {filteredItems.map(item => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  );
}

useCallback Hook for Event Handlers

Memoize event handlers to prevent unnecessary re-renders

Basic useCallback

import React, { useState, useCallback, memo } from 'react';

// Child component that receives callback
const Button = memo(function Button({ onClick, children }) {
  console.log('Button rendered');
  return <button onClick={onClick}>{children}</button>;
});

function ParentComponent() {
  const [count, setCount] = useState(0);
  const [name, setName] = useState('');

  // Without useCallback - creates new function on every render
  const handleClick = () => {
    setCount(c => c + 1);
  };

  // With useCallback - memoizes function
  const memoizedHandleClick = useCallback(() => {
    setCount(c => c + 1);
  }, []); // Empty dependency array since setCount is stable

  const handleNameChange = useCallback((e) => {
    setName(e.target.value);
  }, []);

  return (
    <div>
      <input value={name} onChange={handleNameChange} />
      <p>Count: {count}</p>
      <Button onClick={memoizedHandleClick}>Increment</Button>
    </div>
  );
}

useCallback with Dependencies

import React, { useState, useCallback, memo } from 'react';

const SearchInput = memo(function SearchInput({ onSearch, placeholder }) {
  console.log('SearchInput rendered');
  
  return (
    <input
      type="text"
      placeholder={placeholder}
      onChange={(e) => onSearch(e.target.value)}
    />
  );
});

function SearchComponent() {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);
  const [filter, setFilter] = useState('all');

  // Memoize search function with dependencies
  const handleSearch = useCallback((searchQuery) => {
    setQuery(searchQuery);
    
    // Simulate API call

    );
    
    setResults(filteredResults);
  }, [filter]); // Re-create when filter changes

  // Memoize filter change handler
  const handleFilterChange = useCallback((newFilter) => {
    setFilter(newFilter);
    // Trigger search with current query
    handleSearch(query);
  }, [query, handleSearch]);

  return (
    <div>
      <SearchInput onSearch={handleSearch} placeholder="Search..." />
      <select onChange={(e) => handleFilterChange(e.target.value)}>
        <option value="all">All</option>
        <option value="category1">Category 1</option>
        <option value="category2">Category 2</option>
      </select>
      <ul>
        {results.map(item => (
          <li key={item.id}>{item.name}</li>
        ))}
      </ul>
    </div>
  );
}
