React Fundamentals Examples

Interactive examples demonstrating React core concepts

1. Basic Components & Props

Simple functional components with props and state management.

// Functional component with props
function Welcome({ name }) {
  return <h2>Hello, {name}!</h2>;
}

// Component with state
function Counter() {
  const [count, setCount] = React.useState(0);
  
  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>+</button>
    </div>
  );
}

2. State Management with Hooks

Managing component state with useState and useEffect hooks.

// Multiple state variables
function UserForm() {
  const [name, setName] = React.useState('');
  const [email, setEmail] = React.useState('');
  const [age, setAge] = React.useState(0);
  
  // Effect hook
  React.useEffect(() => {
    document.title = name ? `${name}'s Profile` : 'User Profile';
  }, [name]);
  
  return (
    <form>
      <input value={name} onChange={(e) => setName(e.target.value)} />
      <input value={email} onChange={(e) => setEmail(e.target.value)} />
      <input type="number" value={age} onChange={(e) => setAge(parseInt(e.target.value))} />
    </form>
  );
}

3. Event Handling & Forms

Handling user interactions and form submissions.

// Form handling
function ContactForm() {
  const [formData, setFormData] = React.useState({
    name: '', email: '', message: ''
  });
  
  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('Form submitted:', formData);
  };
  
  const handleChange = (e) => {
    setFormData({
      ...formData,
      [e.target.name]: e.target.value
    });
  };
  
  return (
    <form onSubmit={handleSubmit}>
      <input name="name" value={formData.name} onChange={handleChange} />
      <input name="email" value={formData.email} onChange={handleChange} />
      <textarea name="message" value={formData.message} onChange={handleChange} />
      <button type="submit">Submit</button>
    </form>
  );
}

4. Lists & Keys

Rendering lists of data with proper key management.

// Todo list with keys
function TodoList() {
  const [todos, setTodos] = React.useState([
    { id: 1, text: 'Learn React', completed: false },
    { id: 2, text: 'Build an app', completed: false }
  ]);
  
  const toggleTodo = (id) => {
    setTodos(todos.map(todo =>
      todo.id === id ? { ...todo, completed: !todo.completed } : todo
    ));
  };
  
  return (
    <ul>
      {todos.map(todo => (
        <li key={todo.id} onClick={() => toggleTodo(todo.id)}>
          {todo.text}
        </li>
      ))}
    </ul>
  );
}

5. Conditional Rendering

Conditional rendering based on state and props.

// Conditional rendering
function UserProfile({ user, isLoading }) {
  if (isLoading) {
    return <div>Loading...</div>;
  }
  
  if (!user) {
    return <div>Please log in</div>;
  }
  
  return (
    <div>
      <h2>Welcome, {user.name}!</h2>
      {user.isAdmin && <p>Admin privileges</p>}
    </div>
  );
}

6. Custom Hooks

Creating reusable custom hooks for shared logic.

// Custom hook
function useCounter(initialValue = 0) {
  const [count, setCount] = React.useState(initialValue);
  
  const increment = () => setCount(count + 1);
  const decrement = () => setCount(count - 1);
  const reset = () => setCount(initialValue);
  
  return { count, increment, decrement, reset };
}

// Using custom hook
function Counter() {
  const { count, increment, decrement, reset } = useCounter(0);
  
  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={increment}>+</button>
      <button onClick={decrement}>-</button>
      <button onClick={reset}>Reset</button>
    </div>
  );
}

7. Context API

Sharing data across components with React Context.

// Context creation
const ThemeContext = React.createContext();

// Context provider
function ThemeProvider({ children }) {
  const [theme, setTheme] = React.useState('light');
  
  return (
    <ThemeContext.Provider value={{ theme, setTheme }}>
      {children}
    <ThemeContext.Provider>
  );
}

// Context consumer
function ThemedButton() {
  const { theme, setTheme } = React.useContext(ThemeContext);
  
  return (
    <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
      Toggle Theme
    </button>
  );
}

8. Performance Optimization

Using React.memo, useMemo, and useCallback for optimization.

// Memoized component
const ExpensiveComponent = React.memo(function ExpensiveComponent({ data }) {
  const processedData = React.useMemo(() => {
    return data.map(item => ({
      ...item,
      processed: item.value * 2
    }));
  }, [data]);
  
  return (
    <div>
      {processedData.map(item => (
        <div key={item.id}>{item.processed}</div>
      ))}
    </div>
  );
});

// Memoized callback
function Parent({ items }) {
  const handleClick = React.useCallback((id) => {
    console.log('Clicked:', id);
  }, []);
  
  return (
    <div>
      {items.map(item => (
        <ExpensiveComponent key={item.id} data={item} onClick={handleClick} />
      ))}
    </div>
  );
}
