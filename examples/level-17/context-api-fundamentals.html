Level 17: Context API Fundamentals

Context Creation & Basic Usage

Understanding createContext and Provider components

Basic Context Creation

import React, { createContext, useContext, useState } from 'react';

// 1. Create a context
const ThemeContext = createContext();

// 2. Create a provider component
function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');
  
  const value = {
    theme,
    setTheme,
    isDark: theme === 'dark'
  };
  
  return (
    <ThemeContext.Provider value={value}>
      {children}
    </ThemeContext.Provider>
  );
}

// 3. Create a custom hook for consuming context
function useTheme() {
  const context = useContext(ThemeContext);
  
  if (!context) {
    throw new Error('useTheme must be used within a ThemeProvider');
  }
  
  return context;
}

// 4. Use the context in components
function App() {
  return (
    <ThemeProvider>
      <Header />
      <Main />
    </ThemeProvider>
  );
}

function Header() {
  const { theme, setTheme } = useTheme();
  
  return (
    <header>
      <h1>Current theme: {theme}</h1>
      <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
        Toggle Theme
      </button>
    </header>
  );
}

Context with Default Values

import React, { createContext, useContext } from 'react';

// Context with default value
const UserContext = createContext({
  user: null,
  isAuthenticated: false,
  login: () => {},
  logout: () => {}
});

// Provider with state management
function UserProvider({ children }) {
  const [user, setUser] = useState(null);
  
  const login = (userData) => {
    setUser(userData);
  };
  
  const logout = () => {
    setUser(null);
  };
  
  const value = {
    user,
    isAuthenticated: !!user,
    login,
    logout
  };
  
  return (
    <UserContext.Provider value={value}>
      {children}
    </UserContext.Provider>
  );
}

// Custom hook with error handling
function useUser() {
  const context = useContext(UserContext);
  
  if (context === undefined) {
    throw new Error('useUser must be used within a UserProvider');
  }
  
  return context;
}

// Usage in components
function Profile() {
  const { user, isAuthenticated, logout } = useUser();
  
  if (!isAuthenticated) {
    return <div>Please log in</div>;
  }
  
  return (
    <div>
      <h2>Welcome, {user.name}!</h2>
      <button onClick={logout}>Logout</button>
    </div>
  );
}

Multiple Contexts

import React, { createContext, useContext } from 'react';

// Theme Context
const ThemeContext = createContext();
function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');
  return (
    <ThemeContext.Provider value={{ theme, setTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

// Language Context
const LanguageContext = createContext();
function LanguageProvider({ children }) {
  const [language, setLanguage] = useState('en');
  return (
    <LanguageContext.Provider value={{ language, setLanguage }}>
      {children}
    </LanguageContext.Provider>
  );
}

// Settings Context (combines both)
const SettingsContext = createContext();
function SettingsProvider({ children }) {
  const [settings, setSettings] = useState({
    theme: 'light',
    language: 'en',
    notifications: true
  });
  
  const updateSetting = (key, value) => {
    setSettings(prev => ({ ...prev, [key]: value }));
  };
  
  return (
    <SettingsContext.Provider value={{ settings, updateSetting }}>
      {children}
    </SettingsContext.Provider>
  );
}

// App with multiple providers
function App() {
  return (
    <ThemeProvider>
      <LanguageProvider>
        <SettingsProvider>
          <Header />
          <Main />
        </SettingsProvider>
      </LanguageProvider>
    </ThemeProvider>
  );
}

Context Consumption Patterns

Different ways to consume and use context values

useContext Hook

import React, { useContext } from 'react';

function UserProfile() {
  // Direct context consumption
  const user = useContext(UserContext);
  
  if (!user) {
    return <div>Loading...</div>;
  }
  
  return (
    <div>
      <h2>{user.name}</h2>
      <p>{user.email}</p>
    </div>
  );
}

// Custom hook for better error handling
function useUser() {
  const context = useContext(UserContext);
  
  if (context === undefined) {
    throw new Error('useUser must be used within a UserProvider');
  }
  
  return context;
}

function UserProfileWithHook() {
  const { user, isAuthenticated } = useUser();
  
  if (!isAuthenticated) {
    return <div>Please log in</div>;
  }
  
  return (
    <div>
      <h2>{user.name}</h2>
      <p>{user.email}</p>
    </div>
  );
}

Consumer Component

import React from 'react';

// Using Consumer component (legacy pattern)
function UserProfileConsumer() {
  return (
    <UserContext.Consumer>
      {({ user, isAuthenticated }) => {
        if (!isAuthenticated) {
          return <div>Please log in</div>;
        }
        
        return (
          <div>
            <h2>{user.name}</h2>
            <p>{user.email}</p>
          </div>
        );
      }}
    </UserContext.Consumer>
  );
}

// Higher-order component for context
function withUser(Component) {
  return function WrappedComponent(props) {
    return (
      <UserContext.Consumer>
        {userContext => (
          <Component {...props} user={userContext} />
        )}
      </UserContext.Consumer>
    );
  };
}

// Usage with HOC
const UserProfileHOC = withUser(function UserProfile({ user }) {
  if (!user.isAuthenticated) {
    return <div>Please log in</div>;
  }
  
  return (
    <div>
      <h2>{user.user.name}</h2>
      <p>{user.user.email}</p>
    </div>
  );
});

Context Selectors

import React, { useContext, useMemo } from 'react';

// Context with multiple values
const AppContext = createContext();

function AppProvider({ children }) {
  const [state, setState] = useState({
    user: null,
    theme: 'light',
    language: 'en',
    notifications: []
  });
  
  const value = useMemo(() => ({
    ...state,
    setState,
    // Computed values
    isAuthenticated: !!state.user,
    unreadCount: state.notifications.filter(n => !n.read).length
  }), [state]);
  
  return (
    <AppContext.Provider value={value}>
      {children}
    </AppContext.Provider>
  );
}

// Custom hook with selector
function useAppState(selector) {
  const context = useContext(AppContext);
  
  if (!context) {
    throw new Error('useAppState must be used within AppProvider');
  }
  
  return useMemo(() => selector(context), [context, selector]);
}

// Usage with selectors
function UserInfo() {
  // Only re-renders when user changes
  const user = useAppState(state => state.user);
  
  return <div>{user?.name || 'Guest'}</div>;
}

function ThemeToggle() {
  // Only re-renders when theme changes
  const theme = useAppState(state => state.theme);
  const setState = useAppState(state => state.setState);
  
  const toggleTheme = () => {
    setState(prev => ({
      ...prev,
      theme: prev.theme === 'light' ? 'dark' : 'light'
    }));
  };
  
  return (
    <button onClick={toggleTheme}>
      Current theme: {theme}
    </button>
  );
}
